ConsumerWidget StatsScreen. No constructor parameters needed.

Scaffold with:
- AppBar: title Text('STATS', style: GoogleFonts.sometypeMono(fontWeight: FontWeight.w700)), backgroundColor: Colors.black, foregroundColor: BrdyColors.onSurface, elevation: 0.
- Body: SingleChildScrollView wrapping a Padding(padding: EdgeInsets.all(BrdySpacing.md), child: Column):
  1. Text('HANDICAP TREND', style: GoogleFonts.sometypeMono(fontSize: 12, color: BrdyColors.onSurfaceMuted))
  2. Gap(BrdySpacing.sm)
  3. ref.watch(trendChartProvider).when(
       loading: () => SizedBox(height: 200, child: Center(child: CircularProgressIndicator(color: BrdyColors.accent))),
       error: (_, __) => SizedBox(height: 200, child: Center(child: Text('—', style: GoogleFonts.sometypeMono(color: BrdyColors.onSurfaceMuted)))),
       data: (spots) => DifferentialLineChart(spots: spots),
     )
  4. Gap(BrdySpacing.lg) — use BrdySpacing.lg if defined, else Gap(32)
  5. Text('AVERAGES — ALL ROUNDS', style: GoogleFonts.sometypeMono(fontSize: 12, color: BrdyColors.onSurfaceMuted))
  6. Gap(BrdySpacing.sm)
  7. ref.watch(crossRoundAveragesProvider).when(
       loading: () => const Center(child: CircularProgressIndicator()),
       error: (_, __) => const SizedBox.shrink(),
       data: (averages) => GridView.count(
         crossAxisCount: 2,
         shrinkWrap: true,
         physics: const NeverScrollableScrollPhysics(),
         mainAxisSpacing: BrdySpacing.sm,
         crossAxisSpacing: BrdySpacing.sm,
         childAspectRatio: 2.2,
         children: [
           AverageStatCard(label: 'Score to Par', value: averages == null ? null : _formatScoreToPar(averages.avgScoreToPar)),
           AverageStatCard(label: 'Avg Putts', value: averages == null ? null : averages.avgPutts.toStringAsFixed(1)),
           AverageStatCard(label: 'FIR %', value: averages == null ? null : '${averages.avgFirPercent.toStringAsFixed(0)}%'),
           AverageStatCard(label: 'GIR %', value: averages == null ? null : '${averages.avgGirPercent.toStringAsFixed(0)}%'),
         ],
       ),
     )

Helper _formatScoreToPar: if value >= 0 return '+${value.toStringAsFixed(1)}' else return value.toStringAsFixed(1). Keep as a private function at file level.
